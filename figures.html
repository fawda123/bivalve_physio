toplo <- crsdat %>% 
  mutate(
    Species = case_when(
      Species == 'mussel' ~ 'Mussel', 
      Species == 'scallop' ~ 'Scallop'
    )
  ) %>% 
  unite('fct', Species, `Depth (m)`, sep = ', ') %>% 
  mutate(
    fct = factor(fct, 
                 levels = c('Mussel, 5m', 'Scallop, 5m', 'Mussel, 30m', 'Scallop, 30m'))
  )
  
labs <- levels(toplo$var1)
